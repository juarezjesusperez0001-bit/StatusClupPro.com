    function track(evt, data){ 
      const event = {
        evt, 
        data, 
        t: Date.now(), 
        sessionId: getSessionId(),
        userAgent: navigator.userAgent,
        referrer: document.referrer
      };
      analyticsQueue.push(event); 
      
      // Revenue tracking mejorado
      if(evt === 'affiliate_click') {
        revenueTracking.affiliateClicks++;
        revenueTracking.conversionValue += (data.estimatedValue || 50);
      }
      if(evt === 'premium_interest') {
        revenueTracking.premiumInterest++;
      }
      
      // Envío inmediato para eventos críticos
      if(['subscribe', 'affiliate_click'].includes(evt)) {
        sendAnalytics();
      }
    }
    
    function getSessionId() {
      let sid = sessionStorage.getItem('sc_session');
      if(!sid) {
        sid = 'sc_' + Date.now() + '_' + Math.random().toString(36).substr(2, 9);
        sessionStorage.setItem('sc_session', sid);
      }
      return sid;
    }

    // Filtros mejorados
    function applyFilters(){
      const q = el('#q').value.toLowerCase().trim();
      const c = el('#fCat').value; 
      const n = el('#fNivel').value; 
      const r = el('#fReg').value; 
      const p = parseFloat(el('#fPrecio').value || '0');
      
      const filtered = ITEMS.filter(x=>{
        const matchesSearch = !q || `${x.nombre} ${x.cat} ${x.nivel} ${x.region} ${x.description}`.toLowerCase().includes(q);
        const matchesCategory = !c || x.cat === c;
        const matchesLevel = !n || x.nivel === n;
        const matchesRegion = !r || x.region === r;
        const matchesPrice = !p || x.precio <= p;
        
        return matchesSearch && matchesCategory && matchesLevel && matchesRegion && matchesPrice;
      });
      
      renderItems(filtered);
      track('filter_applied', {
        results: filtered.length,
        filters: {category: c, level: n, region: r, maxPrice: p, search: q}
      });
    }

    // Event listeners para filtros
    ['#q','#fCat','#fNivel','#fReg','#fPrecio'].forEach(sel=>{

      element.addEventListener('input', applyFilters);
      element.addEventListener('change', applyFilters);

    const PAY_KEY = 'sc-membership';
    function isMember(){ 
      const membership = localStorage.getItem(PAY_KEY);
      return membership === 'elite' || membership === 'basic'; 
    }
    function getMembershipLevel(){ 
      return localStorage.getItem(PAY_KEY) || 'none'; 
    }
    
    function openPaywall(){ 
      el('#paywall').classList.add('open');
      track('paywall_opened', {trigger: event.target.dataset.content || 'unknown'});
    }
    
    function closePaywall(){ 
      el('#paywall').classList.remove('open');
      track('paywall_closed', {});
    }

    // Event listeners para paywall (usando delegación de eventos)
    document.addEventListener('click', (e) => {
      if (e.target.matches('[data-open-paywall]') || e.target.closest('[data-open-paywall]')) {
        const trigger = e.target.matches('[data-open-paywall]') ? e.target : e.target.closest('[data-open-paywall]');
        
        if(isMember()){
          alert(`✅ Acceso concedido - Miembro ${getMembershipLevel().toUpperCase()}`);
          track('premium_access_granted', {level: getMembershipLevel()});
        } else { 
          openPaywall(); 
        }
      }
    });

    // Event listeners para cerrar paywall y suscripción a planes (usando delegación)
    document.addEventListener('click', (e) => {
      // Cerrar paywall
      if (e.target.matches('#closePaywall')) {
        closePaywall();
        return;
      }
      
      // Suscripción a planes
      if (e.target.matches('[data-plan]')) {
        const btn = e.target;
        const plan = btn.dataset.plan;
        localStorage.setItem(PAY_KEY, plan);
        closePaywall();
        
        // Simulación de proceso de pago
        btn.innerHTML = 'Procesando...';
        btn.disabled = true;
        
        setTimeout(() => {
          alert(`🎉 ¡Bienvenido a StatusClub ${plan.toUpperCase()}! 

✅ Tu suscripción está activa
📧 Revisa tu email para acceso inmediato  
🎯 Accede ya a todo el contenido premium

Valor de tu plan: ${plan === 'elite' ? '$39/mes' : '$9/mes'}`);
          
          track('subscribe', {plan, value: plan === 'elite' ? 39 : 9});
          btn.innerHTML = plan === 'elite' ? 'Comenzar Elite' : 'Comenzar Básico';
          btn.disabled = false;
        }, 2000);
        return;
      }
    });

    // Newsletter
    el('#formNewsletter')?.addEventListener('submit', (e)=>{
      e.preventDefault();
      const email = el('#email');

      const submitBtn = e.target.querySelector('button[type="submit"]');
      
      if(!email.checkValidity()){ 
        msg.textContent = '❌ Por favor ingresa un email válido.'; 
        email.focus(); 
        return;

      submitBtn.innerHTML = 'Suscribiendo...';
      submitBtn.disabled = true;

        const list = getLeads();

        msg.style.color = 'var(--good)';
        
        // Actualizar contador
        const currentCount = 8950 + list.filter(l => l.type === 'newsletter').length;
        el('#suscritos').textContent = currentCount.toLocaleString('es-MX');
        
        track('lead', {source:'newsletter', email: email.value});
        
        email.value = '';
        submitBtn.innerHTML = 'Suscribirme';
        submitBtn.disabled = false;
